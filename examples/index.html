Programmatic Control

Configuration

const ms4 = new MultiSelect('#select4', {
  placeholder: 'Select days...',
  showClearAll: false
});

// Programmatic methods
window.selectWeekdays = () => {
  ms4.setValue(['mon', 'tue', 'wed', 'thu', 'fri']);
};

window.selectWeekend = () => {
  ms4.setValue(['sat', 'sun']);
};

window.clearSelection = () => {
  ms4.clearAll();
};

üå≤ Nested Options Examples

File System Browser

Configuration

const filesMs = new MultiSelect('#files-select', {
  nestedOptions: true,
  defaultExpanded: true,
  searchEnabled: true,
  expandOnSearch: true,
  placeholder: 'Browse files...'
});

// Expand/collapse methods
window.expandAll = () => filesMs.expandAll();
window.collapseAll = () => filesMs.collapseAll();
